Key Features

AI-Driven Identification

Leverages TensorFlow Lite model trained on regional mushroom datasets for fast, accurate identification. The model works offline and provides detailed information about the identified mushroom.

Implementation:

  • TensorFlow Lite model integration
  • Image preprocessing pipeline
  • Confidence score calculation
  • Multi-angle analysis to improve accuracy

Offline-First Functionality

The app works seamlessly without an internet connection, perfect for foraging in remote areas with no cell service.

Implementation:

  • Hive database for local storage
  • Offline-first architecture pattern
  • Local model inference
  • Sync mechanism when back online

Foraging Site Tracker

GPS pinning and footprint maps allow users to track and revisit their favorite foraging spots. The app records location data, time, and found species.

Implementation:

  • Google Maps integration
  • Geolocator package for precise location
  • Path tracking with Polylines
  • Offline map caching

Safety & Toxicity Warnings

Provides detailed toxicity information and side-by-side comparisons of edible mushrooms with their poisonous lookalikes to prevent misidentification.

Implementation:

  • Visual comparison tool
  • Detailed toxicity database
  • Warning system for dangerous species
  • Expert verification workflow

Community & Expert Verification

Connect with experts and fellow enthusiasts through in-app chat for verification and discussion. Gamified challenges encourage learning and participation.

Implementation:

  • Firebase chat integration
  • Expert verification system
  • Badge and achievement system
  • Community forums

TensorFlow Lite Integration


// infrastructure/services/ml_service.dart
import 'dart:io';
import 'package:flutter/services.dart';
import 'package:image/image.dart' as img;
import 'package:tflite_flutter/tflite_flutter.dart';

class MLService {
  static const String MODEL_PATH = 'assets/ml/mushroom_model.tflite';
  static const String LABELS_PATH = 'assets/ml/mushroom_labels.txt';
  
  late Interpreter _interpreter;
  late List _labels;
  
  Future loadModel() async {
    try {
      // Load TFLite model
      final interpreterOptions = InterpreterOptions();
      
      // Use NNAPI delegate for Android devices
      if (Platform.isAndroid) {
        interpreterOptions.addDelegate(NnApiDelegate());
      }
      
      // Use Metal delegate for iOS devices
      if (Platform.isIOS) {
        interpreterOptions.addDelegate(GpuDelegate());
      }
      
      _interpreter = await Interpreter.fromAsset(
        MODEL_PATH,
        options: interpreterOptions,
      );
      
      // Load labels
      final labelsData = await rootBundle.loadString(LABELS_PATH);
      _labels = labelsData.split('\n');
      
      print('ML model and labels loaded successfully');
    } catch (e) {
      print('Error loading model: $e');
      rethrow;
    }
  }
  
  Future> identifyMushroom(File imageFile) async {
    // Decode and resize image
    final imageBytes = await imageFile.readAsBytes();
    final image = img.decodeImage(imageBytes)!;
    final resizedImage = img.copyResize(image, width: 224, height: 224);
    
    // Convert to binary format expected by the model (float32 buffer)
    final input = _imageToByteListFloat32(resizedImage);
    
    // Output buffer
    final output = List>.filled(
      1,
      List.filled(_labels.length, 0),
    );
    
    // Run inference
    _interpreter.run(input, output);
    
    // Process results
    final result = _processResults(output);
    
    return result;
  }
  
  // Convert image to float32 tensor
  List>>> _imageToByteListFloat32(img.Image image) {
    final tensorImage = List.generate(
      1,
      (_) => List.generate(
        224,
        (y) => List.generate(
          224,
          (x) => List.generate(
            3,
            (c) {
              final pixel = image.getPixel(x, y);
              // Normalize pixel values
              return (c == 0) ? pixel.r / 255.0 : 
                     (c == 1) ? pixel.g / 255.0 : 
                                pixel.b / 255.0;
            },
          ),
        ),
      ),
    );
    return tensorImage;
  }
  
  Map _processResults(List> outputs) {
    // Get the output scores
    final scores = outputs[0];
    
    // Find the highest confidence score and its index
    var maxScore = 0.0;
    var maxIndex = 0;
    
    for (var i = 0; i < scores.length; i++) {
      if (scores[i] > maxScore) {
        maxScore = scores[i];
        maxIndex = i;
      }
    }
    
    // Get top 3 results
    final List> top3Results = [];
    
    // Create a list of (score, index) pairs
    final indexedScores = List>.generate(
      scores.length,
      (i) => {'score': scores[i], 'index': i},
    );
    
    // Sort by score in descending order
    indexedScores.sort((a, b) => (b['score'] as double).compareTo(a['score'] as double));
    
    // Get top 3
    for (var i = 0; i < 3 && i < indexedScores.length; i++) {
      final score = indexedScores[i]['score'] as double;
      final index = indexedScores[i]['index'] as int;
      
      top3Results.add({
        'label': _labels[index],
        'confidence': score,
      });
    }
    
    return {
      'topResult': {
        'label': _labels[maxIndex],
        'confidence': maxScore,
      },
      'top3Results': top3Results,
      'rawScores': scores,
    };
  }
  
  void dispose() {
    _interpreter.close();
  }
}

Location Tracking and Map Integration


// infrastructure/services/location_service.dart
import 'dart:async';
import 'package:geolocator/geolocator.dart';
import 'package:google_maps_flutter/google_maps_flutter.dart';
import 'package:permission_handler/permission_handler.dart';

class LocationService {
  // Stream for real-time location updates
  StreamSubscription? _positionStreamSubscription;
  final _locationController = StreamController.broadcast();
  Stream get locationStream => _locationController.stream;
  
  // Track user path during foraging
  final List _currentPath = [];
  List get currentPath => List.unmodifiable(_currentPath);
  
  // Track if we're currently recording a path
  bool _isRecordingPath = false;
  bool get isRecordingPath => _isRecordingPath;
  
  Future initialize() async {
    await _checkLocationPermission();
  }
  
  Future _checkLocationPermission() async {
    // Check if location services are enabled
    final serviceEnabled = await Geolocator.isLocationServiceEnabled();
    if (!serviceEnabled) {
      return false;
    }
    
    // Check location permission
    var permission = await Geolocator.checkPermission();
    if (permission == LocationPermission.denied) {
      permission = await Geolocator.requestPermission();
      if (permission == LocationPermission.denied) {
        return false;
      }
    }
    
    if (permission == LocationPermission.deniedForever) {
      await openAppSettings();
      return false;
    }
    
    return true;
  }
  
  Future getCurrentLocation() async {
    final hasPermission = await _checkLocationPermission();
    if (!hasPermission) return null;
    
    try {
      return await Geolocator.getCurrentPosition(
        desiredAccuracy: LocationAccuracy.high,
      );
    } catch (e) {
      print('Error getting location: $e');
      return null;
    }
  }
  
  void startLocationTracking() {
    _positionStreamSubscription?.cancel();
    
    _positionStreamSubscription = Geolocator.getPositionStream(
      locationSettings: const LocationSettings(
        accuracy: LocationAccuracy.high,
        distanceFilter: 5, // Update every 5 meters
      ),
    ).listen(
      (Position position) {
        _locationController.add(position);
        
        if (_isRecordingPath) {
          _currentPath.add(LatLng(position.latitude, position.longitude));
        }
      },
      onError: (error) {
        print('Error in location stream: $error');
      },
    );
  }
  
  void stopLocationTracking() {
    _positionStreamSubscription?.cancel();
    _positionStreamSubscription = null;
  }
  
  void startPathRecording() {
    _currentPath.clear();
    _isRecordingPath = true;
    
    // Make sure we're tracking location
    if (_positionStreamSubscription == null) {
      startLocationTracking();
    }
  }
  
  Future> stopPathRecording() async {
    _isRecordingPath = false;
    return _currentPath;
  }
  
  Future saveForagingLocation({
    required String locationName,
    required String notes,
    required List foundSpecies,
    required LatLng coordinates,
    List? path,
  }) async {
    // Implementation would store this in a local database
    // and sync to Firebase when online
    print('Saving foraging location: $locationName at $coordinates');
    
    // This would typically connect to the repository
    // return forageRepository.saveLocation(...);
  }
  
  void dispose() {
    _positionStreamSubscription?.cancel();
    _locationController.close();
  }
}

import 'package:flutter/material.dart';

class MapScreen extends StatefulWidget {

  _MapScreenState createState() => _MapScreenState();

class _MapScreenState extends State {
  GoogleMapController? _mapController;
  final Set _markers = {};
  final Set _polylines = {};
  
  @override
  void initState() {
    super.initState();
    context.read().add(LoadSavedLocationsEvent());
  }
  
  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(
        title: const Text('Foraging Map'),
        actions: [
          BlocBuilder(
            builder: (context, state) {
              final isRecording = state.isRecordingPath;
              return IconButton(
                icon: Icon(
                  isRecording ? Icons.stop_circle : Icons.play_circle,
                  color: isRecording ? Colors.red : Colors.white,
                ),
                onPressed: () {
                  if (isRecording) {
                    context.read().add(StopPathRecordingEvent());
                  } else {
                    context.read().add(StartPathRecordingEvent());
                  }
                },
                tooltip: isRecording ? 'Stop Recording' : 'Start Recording Path',
              );
            },
          ),
        ],
      ),
      body: BlocConsumer(
        listener: (context, state) {
          if (state.status == MapStatus.locationUpdated && 
              state.currentLocation != null) {
            _updateCurrentLocation(state.currentLocation!);
          }
          
          if (state.status == MapStatus.savedLocationsLoaded) {
            _updateSavedLocations(state.savedLocations);
          }
          
          if (state.status == MapStatus.pathUpdated) {
            _updatePathPolyline(state.currentPath);
          }
        },
        builder: (context, state) {
          return GoogleMap(
            initialCameraPosition: CameraPosition(
              target: state.currentLocation ?? const LatLng(37.42796, -122.08574),
              zoom: 15.0,
            ),
            myLocationEnabled: true,
            myLocationButtonEnabled: true,
            markers: _markers,
            polylines: _polylines,
            onMapCreated: (GoogleMapController controller) {
              _mapController = controller;
              // Get current location when map is created
              context.read().add(GetCurrentLocationEvent());
            },
            onLongPress: (LatLng position) {
              _showAddLocationDialog(position);

          ),

    );

    _mapController?.animateCamera(
      CameraUpdate.newLatLng(location),

      _markers.clear();
      
      for (final location in locations) {

          Marker(
            markerId: MarkerId(location.id),
            position: location.coordinates,
            infoWindow: InfoWindow(
              title: location.name,

// domain/models/mushroom.dart
enum Edibility {
  edible,
  inedible,
  poisonous,
  psychoactive,
  unknown,
}

class Mushroom {
  final String id;
  final String commonName;
  final String scientificName;
  final String description;
  final Edibility edibility;
  final String habitat;
  final List traits;
  final List seasons;
  final List lookalikes;
  final double confidence; // Confidence score from identification

  Mushroom({
    required this.id,
    required this.commonName,
    required this.scientificName,
    required this.description,
    required this.edibility,
    required this.habitat,
    required this.traits,
    required this.seasons,
    required this.lookalikes,
    this.confidence = 1.0,
  });

  factory Mushroom.fromJson(Map json) {
    return Mushroom(
      id: json['id'],
      commonName: json['commonName'],
      scientificName: json['scientificName'],
      description: json['description'],
      edibility: _parseEdibility(json['edibility']),
      habitat: json['habitat'],
      traits: List.from(json['traits']),
      seasons: List.from(json['seasons']),
      lookalikes: (json['lookalikes'] as List)
          .map((l) => LookalikeSpecies.fromJson(l))
          .toList(),
      confidence: json['confidence'] ?? 1.0,
    );
  }

  Map toJson() {
    return {
      'id': id,
      'commonName': commonName,
      'scientificName': scientificName,
      'description': description,
      'edibility': edibility.toString().split('.').last,
      'habitat': habitat,
      'traits': traits,
      'seasons': seasons,
      'lookalikes': lookalikes.map((l) => l.toJson()).toList(),
      'confidence': confidence,
    };
  }

  static Edibility _parseEdibility(String value) {
    switch (value.toLowerCase()) {
      case 'edible':
        return Edibility.edible;
      case 'inedible':
        return Edibility.inedible;
      case 'poisonous':
        return Edibility.poisonous;
      case 'psychoactive':
        return Edibility.psychoactive;
      default:
        return Edibility.unknown;
    }
  }
}

class LookalikeSpecies {
  final String id;
  final String commonName;
  final String scientificName;
  final Edibility edibility;
  final String differentiationNotes;

  LookalikeSpecies({
    required this.id,
    required this.commonName,
    required this.scientificName,
    required this.edibility,
    required this.differentiationNotes,
  });

  factory LookalikeSpecies.fromJson(Map json) {
    return LookalikeSpecies(
      id: json['id'],
      commonName: json['commonName'],
      scientificName: json['scientificName'],
      edibility: Mushroom._parseEdibility(json['edibility']),
      differentiationNotes: json['differentiationNotes'],
    );
  }

  Map toJson() {
    return {
      'id': id,
      'commonName': commonName,
      'scientificName': scientificName,
      'edibility': edibility.toString().split('.').last,
      'differentiationNotes': differentiationNotes,
    };
  }
}

class SavedLocation {
  final String id;
  final String name;
  final String notes;
  final DateTime timestamp;
  final LatLng coordinates;
  final List? path;
  final List species;
  final List? photos;

  SavedLocation({
    required this.id,
    required this.name,
    required this.notes,
    required this.timestamp,
    required this.coordinates,
    this.path,
    required this.species,
    this.photos,
  });

  factory SavedLocation.fromJson(Map json) {
    return SavedLocation(
      id: json['id'],
      name: json['name'],
      notes: json['notes'],
      timestamp: DateTime.parse(json['timestamp']),
      coordinates: LatLng(
        json['coordinates']['latitude'],
        json['coordinates']['longitude'],
      ),
      path: json['path'] != null
          ? (json['path'] as List)
              .map((p) => LatLng(p['latitude'], p['longitude']))
              .toList()
          : null,
      species: List.from(json['species']),
      photos: json['photos'] != null ? List.from(json['photos']) : null,
    );
  }

  Map toJson() {
    return {
      'id': id,
      'name': name,
      'notes': notes,
      'timestamp': timestamp.toIso8601String(),
      'coordinates': {
        'latitude': coordinates.latitude,
        'longitude': coordinates.longitude,
      },
      'path': path?.map((p) => {
            'latitude': p.latitude,
            'longitude': p.longitude,
          }).toList(),
      'species': species,
      'photos': photos,
    };
  }
}

  static const String userPreferencesBoxName = 'user_preferences';

    // Initialize Hive
    final appDocumentDir = await getApplicationDocumentsDirectory();
    Hive.init(appDocumentDir.path);
    
    // Register adapters
    Hive.registerAdapter(MushroomAdapter());
    Hive.registerAdapter(EdibilityAdapter());
    Hive.registerAdapter(LookalikeSpeciesAdapter());
    Hive.registerAdapter(SavedLocationAdapter());
    Hive.registerAdapter(LatLngAdapter());
    
    // Open boxes
    await Hive.openBox(mushroomsBoxName);
    await Hive.openBox(savedLocationsBoxName);
    await Hive.openBox(userPreferencesBoxName);
    
    print('Database initialized successfully');
  }
  
  // Mushroom operations
  static Future> getAllMushrooms() async {
    final box = Hive.box(mushroomsBoxName);
    return box.values.toList();
  }
  
  static Future getMushroomById(String id) async {
    final box = Hive.box(mushroomsBoxName);
    return box.get(id);
  }
  
  static Future saveMushroom(Mushroom mushroom) async {
    final box = Hive.box(mushroomsBoxName);
    await box.put(mushroom.id, mushroom);
  }
  
  static Future> searchMushroomsByTraits(List traits) async {
    final box = Hive.box(mushroomsBoxName);
    return box.values.where((mushroom) {
      return traits.every((trait) => 
        mushroom.traits.contains(trait.toLowerCase()));
    }).toList();
  }

    final box = Hive.box(savedLocationsBoxName);
    return box.values.toList();

  static Future saveLocation(SavedLocation location) async {

    await box.put(location.id, location);
  }
  
  static Future deleteLocation(String id) async {
    final box = Hive.box(savedLocationsBoxName);
    await box.delete(id);
